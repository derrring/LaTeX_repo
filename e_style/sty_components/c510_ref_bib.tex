% Biblatex customizations - must use \AtEndPreamble (not begindocument hooks)
% because \renewbibmacro, \DeclareCiteCommand, \AtEveryBibitem are preamble-only.
\makeatletter
\AtEndPreamble{%
    % Check if using authoryear by testing for the style file name
    \ifdefstring{\blx@bbxfile}{authoryear}{%
        % This is authoryear style - apply full customization
        \renewbibmacro*{cite}{%
            \printtext[bibhyperref]{%
                \iffieldundef{shorthand}
                {\ifthenelse{\ifnameundef{labelname}\OR\iffieldundef{labelyear}}
                    {\usebibmacro{cite:label}%
                    \setunit{\printdelim{nonameyeardelim}}}
                    {\printnames{labelname}%
                    \setunit{\printdelim{nameyeardelim}}}%
                \usebibmacro{cite:labeldate+extradate}}
        {\usebibmacro{cite:shorthand}}}}

        \DeclareCiteCommand{\cite}[\mkbibbrackets]
        {\usebibmacro{prenote}}
        {\usebibmacro{citeindex}%
        \usebibmacro{cite}}
        {\multicitedelim}
        {\usebibmacro{postnote}}
    }{%
        % Not authoryear - just change parentheses to brackets
        % without touching any internal macros
        \let\mkbibparens\mkbibbrackets
    }

}
\makeatother

\AtEndPreamble{%
    % Other customizations
    \renewbibmacro{in:}{}
    \AtEveryBibitem{%
        \clearfield{day}%
        \clearfield{endday}%
    }
}
\endinput
