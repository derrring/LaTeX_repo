
% Vertical line
\newcommand\verpar[2]{
    \par\vskip1ex
    \noindent
    \begin{tikzpicture}[
            paragraph/.style = {inner sep=0pt,
            text width=\textwidth-\baselineskip}
        ]
        \node (p1) [paragraph] {#2 \vspace{1pt}};
        \path ([xshift=-0.75\baselineskip] p1.south west) coordinate (p2)
        -- node (p3) [sloped] {\footnotesize #1} (p2 |- p1.north);
        \draw[very thick] (p2) -- (p3.west) (p3.east) -- (p2 |- p1.north);
    \end{tikzpicture}
\vspace{5pt}}
%#1: The label text (appears on the vertical line with rotation)
%#2: The main paragraph content
%usage: \verpar{label}{text}

% Bourbaki dangerous bend symbol (Knuth's manfnt at 10pt)
% \dbend: display/margin symbol (too tall for inline use).
% \dmark{text}: warning paragraph with \dbend in left margin.
% Uses NFSS registration (not \font primitive) for LuaLaTeX compatibility.
% For inline use, consider fourier-orns: \caution (Bourbaki dangerous bend),
% \warning (exclamation triangle), \noway (prohibition sign).
\DeclareFontFamily{U}{manfnt}{}
\DeclareFontShape{U}{manfnt}{m}{n}{<-> manfnt}{}
\ifdefined\dbend
    \renewcommand{\dbend}{{\usefont{U}{manfnt}{m}{n}\char127}}
\else
    \newcommand{\dbend}{{\usefont{U}{manfnt}{m}{n}\char127}}
\fi
\newcommand{\dmark}[1]{%
    \par\smallskip\noindent
    \llap{\dbend\enspace}%
    #1%
    \par\smallskip
}

% --- Inline warning symbols (self-contained tikz, no fourier-orns dependency) ---
% \caution: Bourbaki dangerous bend (triangle + zigzag)
% \warning: alert triangle (triangle + exclamation mark)
% \noway:   prohibition sign (circle + diagonal bar)
\ifdefined\caution\else
\newcommand*{\caution}{\raisebox{0.3ex}{%
    \tikz[baseline=0ex, scale=0.08, line width=0.55pt, line cap=round, line join=round]{
        \draw (0,3.2) -- (-2.8,-1.6) -- (2.8,-1.6) -- cycle;
        \draw (-0.9,-0.1) -- (-0.3,-0.9) -- (0.3,0.9) -- (0.9,0.1);
    }%
}}
\fi
\ifdefined\warning\else
\newcommand*{\warning}{\raisebox{0.3ex}{%
    \tikz[baseline=0ex, scale=0.08, line width=0.55pt, line cap=round, line join=round]{
        \draw (0,3.2) -- (-2.8,-1.6) -- (2.8,-1.6) -- cycle;
        \draw (0,1.6) -- (0,-0.1);
        \fill (0,-0.65) circle (0.28);
    }%
}}
\fi
\ifdefined\noway\else
\newcommand*{\noway}{\raisebox{0.25ex}{%
    \tikz[baseline=0ex, scale=0.08, line width=0.55pt, line cap=round, line join=round]{
        \draw (0,0.8) circle (2.0);
        \draw (-1.41,-0.61) -- (1.41,2.21);
    }%
}}
\fi
