% --- Vertical-bar paragraph ---
% \verpar{label}{text}: paragraph with rotated label on a vertical line.
\newcommand\verpar[2]{
    \par\vskip1ex
    \noindent
    \begin{tikzpicture}[
            paragraph/.style = {inner sep=0pt,
            text width=\textwidth-\baselineskip}
        ]
        \node (p1) [paragraph] {#2 \vspace{1pt}};
        \path ([xshift=-0.75\baselineskip] p1.south west) coordinate (p2)
        -- node (p3) [sloped] {\footnotesize #1} (p2 |- p1.north);
        \draw[very thick] (p2) -- (p3.west) (p3.east) -- (p2 |- p1.north);
    \end{tikzpicture}
\vspace{5pt}}

% =================================================================
% Bourbaki dangerous bend (\dbend) and warning paragraph (\dmark)
% =================================================================
% \dbend:        the manfnt symbol itself (display/margin use only).
% \dmark:        warning paragraph with \dbend in left margin.
%
% The \ifintcbenv flag must be defined HERE (not only in
% e_math_env_deco.sty) so \dmark is safe in classes that don't
% load e_math_env_deco (e.g. article, thesis). e_math_env_deco
% sets the flag true inside its tcolorbox hooks.
% =================================================================

% \ifintcbenv: tcolorbox-context flag. Declared here (default false)
% so any user code that tests it won't crash in classes that don't
% load e_math_env_deco. e_math_env_deco sets it true inside its
% tcolorbox hooks. \dmark itself no longer branches on this flag —
% \marginnote handles the box-escape automatically — but the flag
% is preserved as public API for user customizations.
\newif\ifintcbenv
\intcbenvfalse

% Knuth's manfnt font (NFSS registration, LuaLaTeX-compatible)
\DeclareFontFamily{U}{manfnt}{}
\DeclareFontShape{U}{manfnt}{m}{n}{<-> manfnt}{}
\providecommand{\dbend}{{\usefont{U}{manfnt}{m}{n}\char127}}

% =================================================================
% \dmark — always places \dbend in the LEFT margin, robust across
% all contexts: plain text, mid-paragraph, tcolorbox, minipage,
% lists, columns, one-sided and two-sided documents.
%
% Implementation: TikZ overlay with two pictures per call.
%
%   1. \leavevmode ensures horizontal mode so the inline reference
%      node has a baseline. Required when \dmark is the very first
%      token in a tcolorbox/minipage body.
%
%   2. The inline picture drops a zero-size node containing \strut
%      at the current text position. The strut establishes a normal
%      line baseline and height/depth. The node name embeds a
%      per-call counter so multiple calls do not collide. Internal
%      node names use a private __dmark@ prefix to minimize the
%      chance of clashing with user-named TikZ nodes.
%
%   3. The overlay picture places the \dbend symbol absolutely at
%      the page-left text margin, vertically anchored to the inline
%      node. The x-coordinate is computed from page geometry, with
%      page-parity handling for asymmetric two-sided layouts.
%
% Page-parity reliability: \value{page} is evaluated when the
% overlay picture is shipped out, at which point it correctly
% reflects the page being placed. This makes the \ifodd test
% reliable even for documents with arbitrary page-break behavior.
%
% Engine compatibility: the formula \hoffset + 1in + sidemargin
% handles classes that set \hoffset (e.g. KOMA, some custom classes)
% as well as the standard zero-\hoffset case. PDF engines with
% non-standard \pdfhorigin are not handled here; the marginnote
% package has a more elaborate scheme for that, but it's irrelevant
% for typical LuaLaTeX/pdfLaTeX/XeLaTeX use.
%
% User customization (lengths exposed without @):
%   \dmarkgap   — horizontal gap between symbol and text edge (4pt)
%   \dmarklift  — vertical lift (1\baselineskip)
%   \intcbenvfalse / \intcbenvtrue — preserved public flag (no
%      longer used by \dmark itself; declared above for backward
%      compatibility with user code that may test it)
%
% Requires: TikZ with calc library (loaded by e_core).
% Needs two LaTeX runs for TikZ remember-picture aux-file resolution.
% =================================================================

% --- User-tunable parameters ---
\newlength{\dmarkgap}
\setlength{\dmarkgap}{4pt}

\newlength{\dmarklift}
\AtBeginDocument{\setlength{\dmarklift}{1\baselineskip}}

\makeatletter
\newcounter{__dmark@id}
\setcounter{__dmark@id}{0}

\NewDocumentCommand{\dmark}{}{%
    \leavevmode
    \stepcounter{__dmark@id}%
    % --- Inline reference node at current text baseline ---
    \begin{tikzpicture}[remember picture, overlay, baseline]
        \node [inner sep=0pt, outer sep=0pt]
            (__dmark@ref\arabic{__dmark@id}) {\strut};
    \end{tikzpicture}%
    % --- Symbol placed absolutely at page-left text margin ---
    \begin{tikzpicture}[remember picture, overlay]
        \node [anchor=east, inner sep=0pt]
            at ([xshift=\dimexpr
                  \hoffset + 1in
                  + \ifodd\value{page}\oddsidemargin\else\evensidemargin\fi
                  - \dmarkgap\relax]
                current page.west |- __dmark@ref\arabic{__dmark@id})
            {\raisebox{\dmarklift}{\dbend}};
    \end{tikzpicture}%
    \ % preserve trailing space so "\dmark word" renders with a gap
}
\makeatother
