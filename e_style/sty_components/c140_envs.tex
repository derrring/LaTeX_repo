% =================================================================
% Callout: single aside box (renamed from special_columns).
%
% Usage:
%   \begin{callout}{Title}
%       Body...
%   \end{callout}
%
% Supports \listofcallouts via the 'callouts' tcb list.
% =================================================================
\tcbuselibrary{listings,skins,breakable,fitting,raster,xparse,external}

\tcbset{
    colback=lightgray!15,
    colframe=cyan!40!black,
    fonttitle=\bfseries,
    breakable, enhanced,
    lines before break=5,
}

\newtcolorbox[auto counter, number within=section, list inside=callouts]{callout}[2][]{%
    colback=lightgray!15,
    colframe=cyan!40!black,
    fonttitle=\bfseries,
    breakable, enhanced,
    title=Note: #2, #1%
}

\newcommand{\listofcallouts}{%
    \tcblistof[\section*]{callouts}{List of Callouts}%
}

% --- Attributed quotation (unchanged) ---

        {% End code

            \hspace*{\fill}--- \rmfamily\QuoteAuthor
        \end{minipage}
    \end{flushright}
}
